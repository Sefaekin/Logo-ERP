SELECT 
C.CODE,C.DEFINITION_,
S.CODE AS [SATICI],
MONTH(F.DATE_) AS AY,
COUNT(F.LOGICALREF) AS [FATURA ADEDÝ],
SUM(F.GROSSTOTAL) AS [FATURA TUTARI],
SUM(F.TOTALDISCOUNTS) AS [ÝNDÝRÝM TUTARI],
SUM(F.TOTALDISCOUNTED) AS [KDV MATRAHI],
SUM(F.TOTALVAT) AS [KDV TUTARI],
SUM(F.NETTOTAL) AS [NET TUTAR] 

FROM LG_001_CLCARD C
LEFT JOIN LG_001_01_INVOICE F ON F.CLIENTREF=C.LOGICALREF
LEFT JOIN LG_SLSMAN S ON F.SALESMANREF=S.LOGICALREF
WHERE F.CANCELLED=0
GROUP BY C.CODE,C.DEFINITION_,S.CODE,MONTH(F.DATE_)
