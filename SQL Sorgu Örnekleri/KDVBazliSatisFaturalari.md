SELECT        
ST.DATE_ AS TARÝH, 
CH.FICHENO AS [EVRAK NO], 
C.DEFINITION_ AS [CARÝ ÜNVAN], 
C.TAXNR AS [VERGÝ NO], 
HP.CODE AS [CARÝ MUH. KODU], 
ST.VAT AS KDV, ST.VATMATRAH AS [KDV MATRAHI], 
ST.VATAMNT AS [KDV TUTARI]
FROM            
dbo.LG_100_CLCARD AS C LEFT OUTER JOIN
dbo.LG_100_07_INVOICE AS CH ON C.LOGICALREF = CH.CLIENTREF LEFT OUTER JOIN
dbo.LG_100_07_STLINE AS ST ON CH.LOGICALREF = ST.INVOICEREF LEFT OUTER JOIN
dbo.LG_100_CRDACREF AS CR ON C.LOGICALREF = CR.CARDREF LEFT OUTER JOIN
dbo.LG_100_EMUHACC AS HP ON HP.LOGICALREF = CR.ACCOUNTREF
