SELECT        
CLD.CODE AS [Cari Kodu], 
CLD.DEFINITION_ AS [Cari Hesap Ünvaný], 
SUM(CLT.DEBIT) AS BORC, 
SUM(CLT.CREDIT) AS ALACAK, S
UM(CLT.DEBIT-CLT.CREDIT) AS BAKIYE
FROM dbo.LV_001_01_CLTOTFIL AS CLT INNER JOIN
dbo.LG_001_CLCARD AS CLD ON CLT.CARDREF = CLD.LOGICALREF
WHERE        (CLT.TOTTYP = 1)
GROUP BY CLD.CODE, CLD.DEFINITION_
